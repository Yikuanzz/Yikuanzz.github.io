def SSTF(requests, start):
    seek_sequence = [start]
    total_seek = 0
    while requests:
        closest = min(requests, key=lambda x: abs(x - start))
        total_seek += abs(closest - start)
        start = closest
        seek_sequence.append(closest)
        requests.remove(closest)
