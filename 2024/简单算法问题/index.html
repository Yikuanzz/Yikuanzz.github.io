        self.stack = []
        self.size = size

            print("error")
            return True
        else:
            return False

    def push(self, num):
        self.stack.append(num)
        self.index += 1
    
    def pop(self):
        if self.is_empty():
            return
        else:
            res = self.stack.pop()
            self.index -= 1
            print(res)

    def top(self):
        if self.is_empty():
            return
        else:
            res = self.stack[self.index]
            print(res)
